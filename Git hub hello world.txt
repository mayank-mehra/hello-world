SELECT PACKAGE_ID,OBJECT_NAME,CHANGE_NUMBER,
                    OBJECT_SUFFIX as BASE_OBJECT_SUFFIX,
                    concat(concat(PACKAGE_ID,'/'),OBJECT_NAME)  as  abs_object 
                     FROM CHANGE_ENTRIES 
                     WHERE PACKAGE_ID like 'sial.sapnext%' AND 
                     CHANGE_NUMBER = :P_Changenum;
